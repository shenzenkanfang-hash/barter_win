```rust
//! 订单簿模块 - 20档深度 + 深度指标计算

use rust_decimal::Decimal;
use rust_decimal_macros::dec;

pub struct OrderBook {
    symbol: String,
    bids: Vec<(Decimal, Decimal)>,  // (price, qty)
    asks: Vec<(Decimal, Decimal)>,
    last_update_id: u64,
}

impl OrderBook {
    pub fn new(symbol: String) -> Self {
        Self {
            symbol,
            bids: Vec::with_capacity(20),
            asks: Vec::with_capacity(20),
            last_update_id: 0,
        }
    }

    /// 深度指标: 买盘/卖盘 厚度比率
    pub fn depth_indicator(&self) -> Decimal {
        let bid_depth: Decimal = self.bids.iter().take(20).map(|(_, q)| q).sum();
        let ask_depth: Decimal = self.asks.iter().take(20).map(|(_, q)| q).sum();
        if ask_depth > dec!(0) {
            bid_depth / ask_depth
        } else {
            dec!(1)
        }
    }

    pub fn update(&mut self, last_update_id: u64, bids: Vec<(Decimal, Decimal)>, asks: Vec<(Decimal, Decimal)>) {
        // 确保 update_id 递增
        if last_update_id > self.last_update_id {
            self.bids = bids;
            self.asks = asks;
            self.last_update_id = last_update_id;
        }
    }

    pub fn last_update_id(&self) -> u64 {
        self.last_update_id
    }

    pub fn symbol(&self) -> &str {
        &self.symbol
    }

    pub fn best_bid(&self) -> Option<Decimal> {
        self.bids.first().map(|(p, _)| *p)
    }

    pub fn best_ask(&self) -> Option<Decimal> {
        self.asks.first().map(|(p, _)| *p)
    }

    /// 订单簿总档数
    pub fn depth(&self) -> (usize, usize) {
        (self.bids.len(), self.asks.len())
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_depth_indicator() {
        let mut ob = OrderBook::new("BTCUSDT".to_string());
        ob.update(1,
            vec![(dec!(100), dec!(10)), (dec!(99), dec!(20))],
            vec![(dec!(101), dec!(15)), (dec!(102), dec!(25))],
        );

        let indicator = ob.depth_indicator();
        // bid_depth = 10 + 20 = 30
        // ask_depth = 15 + 25 = 40
```

```rust
//! 波动率检测器 - 1m O-C 和 15m Close-Close 变化率

use rust_decimal::Decimal;
use rust_decimal_macros::dec;
use crate::kline::KLineSynthesizer;
use crate::types::{KLine, VolatilityStats};

pub struct VolatilityDetector {
    /// 1m K线合成器
    kline_1m: KLineSynthesizer,
    /// 15m K线合成器 (内部合成)
    kline_15m: KLineSynthesizer,
    /// 15m 滑动窗口 (保留最近2根)
    kline_15m_window: Vec<KLine>,
    /// 阈值: 1m 3%
    threshold_1m: Decimal,
    /// 阈值: 15m 13%
    threshold_15m: Decimal,
}

impl VolatilityDetector {
    pub fn new(symbol: String) -> Self {
        Self {
            kline_1m: KLineSynthesizer::new(symbol.clone(), crate::types::Period::Minute(1)),
            kline_15m: KLineSynthesizer::new(symbol, crate::types::Period::Minute(15)),
            kline_15m_window: Vec::with_capacity(2),
            threshold_1m: dec!(0.03),
            threshold_15m: dec!(0.13),
        }
    }

    /// 更新并检测波动率
    /// 返回 VolatilityStats
    pub fn update(&mut self, price: Decimal, timestamp: chrono::DateTime<chrono::Utc>) -> VolatilityStats {
        // 1. 更新 1m K线
        let tick = crate::types::Tick {
            symbol: self.kline_1m.symbol.clone(),
            price,
            qty: dec!(1),
            timestamp,
        };
        let completed_1m = self.kline_1m.update(&tick);

        // 2. 计算 1m O-C 变化率
        let vol_1m = self.calc_1m_volatility();

        // 3. 更新 15m K线 (内部合成)
        if let Some(ref completed) = completed_1m {
            // 用完成的 1m K线更新 15m
            let kline_15m_completed = self.kline_15m.update(&tick);
            // 如果 15m K线完成，更新窗口
            if kline_15m_completed.is_some() {
                self.update_15m_window_internal();
            }
        }

        // 4. 计算 15m Close-Close 变化率
        let vol_15m = self.calc_15m_volatility();

        // 5. 判断是否高波动
        let is_high = vol_1m >= self.threshold_1m || vol_15m >= self.threshold_15m;

        VolatilityStats {
            is_high_volatility: is_high,
            vol_1m,
            vol_15m,
        }
    }

    /// 计算 1m O-C 变化率
    fn calc_1m_volatility(&self) -> Decimal {
        let kline = match self.kline_1m.current_kline() {
            Some(k) => k,
            None => return dec!(0),
        };

        if kline.open > dec!(0) {
            (kline.close - kline.open).abs() / kline.open
        } else {
            dec!(0)
        }
    }

    /// 计算 15m Close-Close 变化率
    fn calc_15m_volatility(&self) -> Decimal {
        if self.kline_15m_window.len() < 2 {
            return dec!(0);
        }

        let prev = &self.kline_15m_window[0];
        let curr = &self.kline_15m_window[1];

        if prev.close > dec!(0) {
            (curr.close - prev.close).abs() / prev.close
        } else {
            dec!(0)
        }
    }

    /// 内部更新 15m 窗口
    fn update_15m_window_internal(&mut self) {
        if let Some(kline) = self.kline_15m.current_kline() {
            if self.kline_15m_window.len() >= 2 {
                self.kline_15m_window.remove(0);
            }
            self.kline_15m_window.push(kline.clone());
        }
    }

    pub fn thresholds(&self) -> (Decimal, Decimal) {
        (self.threshold_1m, self.threshold_15m)
    }
}

#[cfg(test)]
mod tests {
    use super::*;
    use chrono::Utc;

    #[test]
    fn test_volatility_detection() {
        let mut detector = VolatilityDetector::new("BTCUSDT".to_string());
        let now = Utc::now();

        // 正常价格
        let stats = detector.update(dec!(50000), now);
        assert!(!stats.is_high_volatility);

        // 1m O-C > 3% 触发
        // ... (需要构造大波动场景)
    }
}
```

```rust
//! 品种注册中心 - Redis 存储 + 订阅管理

use fnv::FnvHashSet;
use redis::AsyncCommands;
use std::sync::Arc;
use tokio::sync::RwLock;

pub struct SymbolRegistry {
    redis: redis::aio::ConnectionManager,
    trading_symbols: RwLock<FnvHashSet<String>>,
    last_update: std::time::Instant,
    update_interval: std::time::Duration,
}

impl SymbolRegistry {
    pub async fn new(redis_url: &str) -> Result<Self, crate::error::MarketError> {
        let client = redis::Client::open(redis_url)
            .map_err(|e| crate::error::MarketError::RedisError(e.to_string()))?;
        let conn = client.get_connection_manager()
            .map_err(|e| crate::error::MarketError::RedisError(e.to_string()))?;

        Ok(Self {
            redis: conn,
            trading_symbols: RwLock::new(FnvHashSet::default()),
            last_update: std::time::Instant::now(),
            update_interval: std::time::Duration::from_secs(120),  // 2分钟
        })
    }

    /// 从 Binance 获取交易对信息并更新 Redis
    pub async fn update_symbols(&mut self) -> Result<(), crate::error::MarketError> {
        // 1. 获取 exchangeInfo
        let client = reqwest::Client::new();
        let resp = client
            .get("https://fapi.binance.com/fapi/v1/exchangeInfo")
            .send()
            .await
            .map_err(|e| crate::error::MarketError::NetworkError(e.to_string()))?;

        let info: serde_json::Value = resp
            .json()
            .await
            .map_err(|e| crate::error::MarketError::ParseError(e.to_string()))?;

        // 2. 解析并存储
        if let Some(symbols) = info.get("symbols").and_then(|s| s.as_array()) {
            let mut new_symbols = FnvHashSet::default();

            for symbol_info in symbols {
                if let (Some(symbol), Some(status)) =
                    (symbol_info.get("symbol").and_then(|s| s.as_str()),
                     symbol_info.get("status").and_then(|s| s.as_str()))
                {
                    if status == "TRADING" {
                        // 存储到 Redis Hash
                        let key = format!("exchangeInfo:{}", symbol);
                        let json = symbol_info.to_string();
                        let _: () = self.redis.hset("exchangeInfo", symbol, &json).await
                            .map_err(|e| crate::error::MarketError::RedisError(e.to_string()))?;

                        new_symbols.insert(symbol.to_string());
                    }
                }
            }

            // 更新内存中的交易品种
            let mut symbols_guard = self.trading_symbols.write().await;
            *symbols_guard = new_symbols;
        }

        self.last_update = std::time::Instant::now();
        Ok(())
    }

    /// 检查是否需要更新
    pub fn needs_update(&self) -> bool {
        self.last_update.elapsed() >= self.update_interval
    }

    /// 获取所有交易品种
    pub async fn get_trading_symbols(&self) -> FnvHashSet<String> {
        self.trading_symbols.read().await.clone()
    }

    /// 获取品种规则
    pub async fn get_symbol_info(&self, symbol: &str) -> Option<String> {
        let info: Option<String> = self.redis.hget("exchangeInfo", symbol).await.ok()?;
        info
    }
}
```

```rust
//! DataFeeder - 核心数据分发器，统一管理 4 个 WebSocket 连接

use crate::binance_ws::{BinanceWsConnector, BinanceKlineMsg, BinanceDepthMsg};
use crate::orderbook::OrderBook;
use crate::volatility::VolatilityDetector;
use crate::symbol_registry::SymbolRegistry;
use crate::types::{KLine, Tick};
use futures_util::StreamExt;
use std::collections::FnvHashMap;
use tokio::sync::broadcast;
use tokio::time::{sleep, Duration};

const PAGINATE_SIZE: usize = 50;
const PAGINATE_INTERVAL_MS: u64 = 500;

pub struct DataFeeder {
    // 4 个 WebSocket 连接
    kline_1m_conn_1: Option<BinanceWsConnector>,
    kline_1m_conn_2: Option<BinanceWsConnector>,
    kline_1d_conn: Option<BinanceWsConnector>,
    depth_conn: Option<BinanceWsConnector>,

    // 广播通道
    broadcaster: broadcast::Sender<DataMessage>,

    // 波动率检测器 (按品种)
    volatility_detectors: FnvHashMap<String, VolatilityDetector>,

    // 订单簿 (按品种)
    orderbooks: FnvHashMap<String, OrderBook>,

    // Depth 订阅管理器
    depth_subscribed: FnvHashSet<String>,
    default_depth_symbol: String,

    // 品种注册
    symbol_registry: SymbolRegistry,

    // 状态
    is_initialized: bool,
    last_data_time: std::time::Instant,
}

pub enum DataMessage {
    KLine1m { symbol: String, kline: KLine },
    KLine1d { symbol: String, kline: KLine },
    Depth { symbol: String, orderbook: OrderBook },
}

impl DataFeeder {
    pub async fn new(redis_url: &str) -> Result<Self, crate::error::MarketError> {
        let (tx, _) = broadcast::channel(1024);

        Ok(Self {
            kline_1m_conn_1: None,
            kline_1m_conn_2: None,
            kline_1d_conn: None,
            depth_conn: None,
            broadcaster: tx,
            volatility_detectors: FnvHashMap::default(),
            orderbooks: FnvHashMap::default(),
            depth_subscribed: FnvHashSet::default(),
            default_depth_symbol: "BTCUSDT".to_string(),
            symbol_registry: SymbolRegistry::new(redis_url).await?,
            is_initialized: false,
            last_data_time: std::time::Instant::now(),
        })
    }

    /// 启动 DataFeeder
    pub async fn start(&mut self) -> Result<(), crate::error::MarketError> {
        // 1. 更新品种列表
        self.symbol_registry.update_symbols().await?;

        // 2. 建立 4 个连接
        self.kline_1m_conn_1 = Some(BinanceWsConnector::new("1m_group1"));
        self.kline_1m_conn_2 = Some(BinanceWsConnector::new("1m_group2"));
        self.kline_1d_conn = Some(BinanceWsConnector::new("1d"));
        self.depth_conn = Some(BinanceWsConnector::new("depth"));

        // 3. 分片订阅
        let symbols = self.symbol_registry.get_trading_symbols().await;
        let symbols_vec: Vec<String> = symbols.into_iter().collect();

        // 分片订阅 1m (分两组)
        let mid = symbols_vec.len() / 2;
        self.subscribe_kline_batch(&symbols_vec[..mid], "1m").await?;
        self.subscribe_kline_batch(&symbols_vec[mid..], "1m").await?;

        // 订阅 1d
        self.subscribe_kline_batch(&symbols_vec, "1d").await?;

        // 订阅默认 Depth
        self.subscribe_depth(&[self.default_depth_symbol.clone()]).await?;

        // 4. 启动数据处理任务
        tokio::spawn(async move {
            self.process_messages().await;
        });

        // 5. 启动品种更新任务
        tokio::spawn(async move {
            self.symbol_update_loop().await;
        });

        Ok(())
    }

    /// 分片订阅 K线
    async fn subscribe_kline_batch(&self, symbols: &[String], interval: &str) -> Result<(), crate::error::MarketError> {
        for chunk in symbols.chunks(PAGINATE_SIZE) {
            let streams: Vec<String> = chunk.iter()
                .map(|s| format!("{}@kline_{}", s.to_lowercase(), interval))
                .collect();

            // 发送订阅消息
            // self.kline_conn.send_subscribe(&streams).await?;

            sleep(Duration::from_millis(PAGINATE_INTERVAL_MS)).await;
        }
        Ok(())
    }

    /// 订阅 Depth
    async fn subscribe_depth(&mut self, symbols: &[String]) -> Result<(), crate::error::MarketError> {
        for symbol in symbols {
            let stream = format!("{}@depth@100ms", symbol.to_lowercase());
            // self.depth_conn.send_subscribe(&stream).await?;
            self.depth_subscribed.insert(symbol.clone());
        }
        Ok(())
    }

    /// 处理消息循环
    async fn process_messages(&mut self) {
        loop {
            // 处理 1m K线消息
            if let Some(conn) = &mut self.kline_1m_conn_1 {
                while let Some(msg) = conn.next_message().await {
                    self.handle_kline_1m(msg).await;
                }
            }
            sleep(Duration::from_millis(1)).await;
        }
    }

    /// 处理 1m K线消息
    async fn handle_kline_1m(&mut self, msg: BinanceKlineMsg) {
        self.last_data_time = std::time::Instant::now();

        let symbol = msg.symbol.clone();

        // 创建/获取波动率检测器
        let detector = self.volatility_detectors.entry(symbol.clone())
            .or_insert_with(|| VolatilityDetector::new(symbol.clone()));

        // 转换为 KLine
        let kline: KLine = msg.into();
        let price = kline.close;
        let timestamp = kline.timestamp;

        // 更新波动率检测器
        let stats = detector.update(price, timestamp);

        // 广播 1m K线
        let _ = self.broadcaster.send(DataMessage::KLine1m {
            symbol: symbol.clone(),
            kline,
        });

        // 如果是高波动，订阅 Depth
        if stats.is_high_volatility && !self.depth_subscribed.contains(&symbol) {
            self.subscribe_depth(&[symbol.clone()]).await.ok();
        }
    }

    /// 品种更新循环
    async fn symbol_update_loop(&mut self) {
        loop {
            sleep(Duration::from_secs(120)).await;

            if self.symbol_registry.needs_update() {
                if let Err(e) = self.symbol_registry.update_symbols().await {
                    tracing::error!("品种更新失败: {}", e);
                }
            }
        }
    }

    /// 订阅
    pub fn subscribe(&self) -> broadcast::Receiver<DataMessage> {
        self.broadcaster.subscribe()
    }
}
```

```rust
//! K线 Redis 持久化 - 1m/15m/1d 历史K线存储

const KLINE_1M_MAX: isize = 320;
const KLINE_15M_MAX: isize = 100;
const KLINE_1D_MAX: isize = 100;

pub struct KlinePersistence {
    redis: redis::aio::ConnectionManager,
}

impl KlinePersistence {
    pub async fn new(redis_url: &str) -> Result<Self, crate::error::MarketError> {
        let client = redis::Client::open(redis_url)
            .map_err(|e| crate::error::MarketError::RedisError(e.to_string()))?;
        let conn = client.get_connection_manager()
            .map_err(|e| crate::error::MarketError::RedisError(e.to_string()))?;

        Ok(Self { redis: conn })
    }

    /// 存储 1m K线
    pub async fn push_1m(&mut self, symbol: &str, kline_json: &str) -> Result<(), crate::error::MarketError> {
        let key = format!("kline:1m:{}", symbol);
        let _: () = self.redis.lpush(&key, kline_json).await
            .map_err(|e| crate::error::MarketError::RedisError(e.to_string()))?;
        let _: () = self.redis.ltrim(&key, 0, KLINE_1M_MAX - 1).await
            .map_err(|e| crate::error::MarketError::RedisError(e.to_string()))?;
        Ok(())
    }

    /// 存储 15m K线
    pub async fn push_15m(&mut self, symbol: &str, kline_json: &str) -> Result<(), crate::error::MarketError> {
        let key = format!("kline:15m:{}", symbol);
        let _: () = self.redis.lpush(&key, kline_json).await
            .map_err(|e| crate::error::MarketError::RedisError(e.to_string()))?;
        let _: () = self.redis.ltrim(&key, 0, KLINE_15M_MAX - 1).await
            .map_err(|e| crate::error::MarketError::RedisError(e.to_string()))?;
        Ok(())
    }

    /// 存储 1d K线
    pub async fn push_1d(&mut self, symbol: &str, kline_json: &str) -> Result<(), crate::error::MarketError> {
        let key = format!("kline:1d:{}", symbol);
        let _: () = self.redis.lpush(&key, kline_json).await
            .map_err(|e| crate::error::MarketError::RedisError(e.to_string()))?;
        let _: () = self.redis.ltrim(&key, 0, KLINE_1D_MAX - 1).await
            .map_err(|e| crate::error::MarketError::RedisError(e.to_string()))?;
        Ok(())
    }

    /// 读取 1m K线
    pub async fn get_1m(&mut self, symbol: &str) -> Result<Vec<String>, crate::error::MarketError> {
        let key = format!("kline:1m:{}", symbol);
        let klines: Vec<String> = self.redis.lrange(&key, 0, -1).await
            .map_err(|e| crate::error::MarketError::RedisError(e.to_string()))?;
        Ok(klines)
    }
}
```

```rust
impl BinanceWsConnector {
    /// 发送订阅消息
    pub async fn subscribe(&mut self, streams: &[String]) -> Result<(), MarketError> {
        let msg = serde_json::json!({
            "method": "SUBSCRIBE",
            "params": streams,
            "id": chrono::Utc::now().timestamp_millis()
        });

        self.send_json(&msg).await
    }

    /// 发送退订消息
    pub async fn unsubscribe(&mut self, streams: &[String]) -> Result<(), MarketError> {
        let msg = serde_json::json!({
            "method": "UNSUBSCRIBE",
            "params": streams,
            "id": chrono::Utc::now().timestamp_millis()
        });

        self.send_json(&msg).await
    }

    async fn send_json(&mut self, msg: &serde_json::Value) -> Result<(), MarketError> {
        let text = serde_json::to_string(msg)
            .map_err(|e| MarketError::SerializeError(e.to_string()))?;

        self.ws_stream.write_message(Message::Text(text.into()))
            .await
            .map_err(|e| MarketError::WebSocketError(e.to_string()))?;

        Ok(())
    }
}
```
